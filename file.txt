CREATE TABLE Cars(
    code_car INT NOT NULL PRIMARY KEY,
    code_driver INT NOT NULL,
    code_marks INT NOT NULL,
    number VARCHAR(255) NOT NULL,
    vin VARCHAR(255) NOT NULL,
    vin_engine VARCHAR(255) NOT NULL,
    number_techpass VARCHAR(255) NOT NULL,
    date_release DATE NOT NULL,
    date_reg DATE NOT NULL,
    color VARCHAR(255) NOT NULL,
    description TEXT NOT NULL,
    id_employee BIGINT NOT NULL
);
CREATE TABLE Ranks(
    rank_code INT NOT NULL PRIMARY KEY,
    rank_names VARCHAR(255) NOT NULL,
    allowance VARCHAR(255) NOT NULL,
    duties VARCHAR(255) NOT NULL,
    requirements VARCHAR(255) NOT NULL
);
CREATE TABLE technical_inspection(
    code_car INT NOT NULL PRIMARY KEY,
    description BIGINT NOT NULL,
    date_tech BIGINT NOT NULL
);
CREATE TABLE Car_brands(
    code_marks INT NOT NULL PRIMARY KEY,
    names VARCHAR(255) NOT NULL,
    retailer VARCHAR(255) NOT NULL,
    country VARCHAR(255) NOT NULL,
    date_start DATE NOT NULL,
    date_end DATE NOT NULL,
    specifications TEXT NOT NULL,
    category VARCHAR(255) NOT NULL,
    description TEXT NOT NULL
);
CREATE TABLE Drivers(
    code_driver INT NOT NULL PRIMARY KEY,
    fio VARCHAR(255) NOT NULL,
    date_birthday DATE NOT NULL,
    adres VARCHAR(255) NOT NULL,
    passport VARCHAR(255) NOT NULL,
    driver_number INT NOT NULL,
    date_of_issue DATE NOT NULL,
    date_end DATE NOT NULL,
    cat CHAR(255) NOT NULL,
    description TEXT NOT NULL,
    id_employee INT NOT NULL
);
CREATE TABLE Stolen_cars(
    date_theft INT NOT NULL PRIMARY KEY,
    date_appeal DATE NOT NULL,
    code_car INT NOT NULL,
    code_driver INT NOT NULL,
    circumstances TEXT NOT NULL,
    mark_find INT NOT NULL,
    date_find DATE NOT NULL,
    id_employee INT NOT NULL
);
CREATE TABLE Employees(
    Employee_ID INT NOT NULL PRIMARY KEY,
    FIO_OF_EMPLOYEE TEXT NOT NULL,
    OLD_OF_EMPLOYEE INT NOT NULL,
    Gender_OF_EMPLOYEE TEXT NOT NULL,
    ADRESS_OF_EMPLOYEE VARCHAR(255) NOT NULL,
    Phone VARCHAR(255) NOT NULL,
    PASSPORT_OF_EMPLOYEE VARCHAR(255) NOT NULL,
    POSITION_ID INT NOT NULL,
    RANK_CODE INT NOT NULL
);
CREATE TABLE Positions(
    id_position INT NOT NULL PRIMARY KEY,
    name_position VARCHAR(255) NOT NULL,
    salary DOUBLE(8, 2) NOT NULL,
    duties VARCHAR(255) NOT NULL,
    requirements TEXT NOT NULL
);
ALTER TABLE
    Cars ADD CONSTRAINT cars_code_marks_foreign FOREIGN KEY(code_marks) REFERENCES Car_brands(code_marks);
ALTER TABLE
    Employees ADD CONSTRAINT employees_position_id_foreign FOREIGN KEY(POSITION_ID) REFERENCES Positions(id_position);
ALTER TABLE
    Cars ADD CONSTRAINT cars_id_employee_foreign FOREIGN KEY(id_employee) REFERENCES Employees(Employee_ID);
ALTER TABLE
    Cars ADD CONSTRAINT cars_code_driver_foreign FOREIGN KEY(code_driver) REFERENCES Drivers(code_driver);
ALTER TABLE
    Employees ADD CONSTRAINT employees_rank_code_foreign FOREIGN KEY(RANK_CODE) REFERENCES Ranks(rank_code);
ALTER TABLE
    Drivers ADD CONSTRAINT drivers_id_employee_foreign FOREIGN KEY(id_employee) REFERENCES Employees(Employee_ID);
ALTER TABLE
    technical_inspection ADD CONSTRAINT technical_inspection_code_car_foreign FOREIGN KEY(code_car) REFERENCES Cars(code_car);
ALTER TABLE
    Stolen_cars ADD CONSTRAINT stolen_cars_code_car_foreign FOREIGN KEY(code_car) REFERENCES Cars(code_car);
